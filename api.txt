APIs
----

API = http://ec2-54-169-173-47.ap-southeast-1.compute.amazonaws.com
repoURL = https://github.com/lnxz/GitGuard

/authors repo
e.g. hostname/authors?repo=repoURL

/authorsAdditionsDeletions repo
e.g. hostname/authorsAdditionsDeletions?repo=repoURL

/commits repo author
e.g. hostname/commits?repo=repoURL&author=Siro

/commitCount repo
e.g. hostname/commitCount?repo=repoUrl

/files repo
e.g. hostname/files?repo=repoURL

/fileStats repo topN
e.g. hostname/fileStats?repo=repoURL&topN=10

/codes repo branch
e.g. hostname/codes?repo=repoURL&branch=master&file=pages/compare.html

/whosYourDaddy repo lineStart lineEnd file
e.g. hostname/whosYourDaddy?repo=repoURL&startLine=1&endLine=3&file=pages/compare.html
